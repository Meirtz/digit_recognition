    y_gen = np.array([np.zeros((X_len, n_class), dtype=np.uint8) for i in range(n_len)])
    # TODO: 1~5
    for i in range(X_len):
        n = random.randint(1,5)
        #print(n)
        for j in range(n):
            index = random.randint(0, 9)
            X_gen[i,:, j*width:(j+1)*width, 0] = X_raw[index]
            y_gen[j][i,y_raw[index]] = 1
        for j in range(n, 5):
            y_gen[j][i,10] = 1
    
    print ("X_gen:", X_gen.shape)
    print("Y_gen", y_gen.shape)
    
    return X_gen, y_gen
